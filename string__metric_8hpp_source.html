156 template <typename Sentence1, typename Sentence2>
157 std::size_t levenshtein(const Sentence1& s1, const Sentence2& s2,
158  LevenshteinWeightTable weights = {1, 1, 1},
159  std::size_t max = std::numeric_limits<std::size_t>::max())
160 {
161  auto sentence1 = common::to_string_view(s1);
162  auto sentence2 = common::to_string_view(s2);
163 
164  if (weights.insert_cost == weights.delete_cost) {
165  /* when insertions + deletions operations are free there can not be any edit distance */
166  if (weights.insert_cost == 0) {
167  return 0;
168  }
169 
170  /* uniform Levenshtein multiplied with the common factor */
171  if (weights.insert_cost == weights.replace_cost) {
172  // max can make use of the common divisor of the three weights
173  const std::size_t new_max = max / weights.insert_cost + (std::size_t)(max % weights.insert_cost != 0);
174  const std::size_t distance = detail::levenshtein(sentence1, sentence2, new_max) * weights.insert_cost;
175  return (distance <= max) ? distance : (std::size_t)-1;
176  }
177  /*
178  * when replace_cost >= insert_cost + delete_cost no substitutions are performed
179  * therefore this can be implemented as InDel distance multiplied with the common factor
180  */
181  else if (weights.replace_cost >= weights.insert_cost + weights.delete_cost) {
182  // max can make use of the common divisor of the three weights
183  const std::size_t new_max = max / weights.insert_cost + (std::size_t)(max % weights.insert_cost != 0);
184  const std::size_t distance = detail::weighted_levenshtein(sentence1, sentence2, new_max) * weights.insert_cost;
185  return (distance <= max) ? distance : (std::size_t)-1;
186  }
187  }
188 
189  return detail::generic_levenshtein(sentence1, sentence2, weights, max);
190 }

393 template <typename Sentence1, typename Sentence2>
394 std::size_t hamming(const Sentence1& s1, const Sentence2& s2,
395  std::size_t max = std::numeric_limits<std::size_t>::max())
396 {
397  auto sentence1 = common::to_string_view(s1);
398  auto sentence2 = common::to_string_view(s2);
399 
400  if (sentence1.size() != sentence2.size()) {
401  throw std::invalid_argument("s1 and s2 are not the same length.");
402  }
403 
404  std::size_t hamm = 0;
405 
406  for (std::size_t i = 0; i < sentence1.length(); i++) {
407  if (common::mixed_sign_unequal(sentence1[i], sentence2[i])) {
408  hamm++;
409  }
410  }
411 
412  return hamm > max ? (std::size_t)-1 : hamm;
413 }
